Table of n Program



    #include <iostream>
    using namespace std;

    int main()
    {
    int num;
    cout << "Enter number";
    cin>>num;

    cout<<"printing the table below";
    for(int i=1; i<=10; i++)
    {
        cout<<num<<" X " <<i<<" = "<<num*i<<endl;
    }
    return 0;
    }

        5
        printing the table below
